use Cro::HTTP::Router;
use Cro::HTTP::Server;
use Cro::HTTP::Client;

# the tiny service from the previous page
my $application = route {
    get -> 'hello' { content 'text/plain', 'Hello from Cro!'; }
}

    :host('127.0.0.1'), :port(8080), :application($application),
);
$server.start;

# now be the client and call it

$server.stop;
